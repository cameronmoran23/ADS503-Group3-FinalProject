# Create train/test partitions for each of the subsets
set.seed(123)
training_set_list <- list()
test_set_list <- list()
for (subset in bank_fraud_df_list) {
    curr_train_rows <- createDataPartition(subset[[target_name]], p = 0.8, list = FALSE)
    curr_train_data <- subset[curr_train_rows, ]
    curr_test_data <- subset[-curr_train_rows, ]
    training_set_list <- append(training_set_list, list(curr_train_data))
    test_set_list <- append(test_set_list, list(curr_test_data))
}

ctrl <- trainControl(method = "cv",
                     summaryFunction = twoClassSummary,
                     classProbs = TRUE,
                     savePredictions = TRUE)

lr_models <- list()
testResults_list <- list()
subset_labels <- c("1%", "5%", "10%")

for (i in 1:3) {
    print(paste0("Logistic Regression — ", subset_labels[i], " subset"))

    trainData <- training_set_list[[i]]
    testData <- test_set_list[[i]]
    predictor_cols <- predictors_name_list

    set.seed(476)
    lrFit <- train(x = trainData[, predictor_cols],
                   y = trainData$is_fraud,
                   method = "glm",
                   metric = "ROC",
                   trControl = ctrl)
    print(lrFit)
    lr_models[[i]] <- lrFit

    # Confusion matrix from cross-validated predictions
    lrCM <- confusionMatrix(lrFit, norm = "none")
    print(lrCM)

    # ROC curve from cross-validated predictions
    lrRoc <- roc(response = lrFit$pred$obs,
                predictor = lrFit$pred$fraud,
                levels = rev(levels(lrFit$pred$obs)))
    plot(lrRoc, legacy.axes = TRUE, main = paste("Logistic Regression ROC -", subset_labels[i]))
    print(paste0("AUC for the ", subset_labels[i], " subset: ", round(lrRoc$auc, 3)))

    # Variable importance
    lrImp <- varImp(lrFit, scale = FALSE)
    plot(lrImp)

    # Saving test-set predictions for the later model comparison step
    testResults_list[[i]] <- data.frame(obs = testData$is_fraud,
                                         LR = predict(lrFit, testData[, predictor_cols]))
}
